# APT Packages for RC Corridor Navigation Sim (Ubuntu 22.04, ROS 2 Humble)

## Core ROS 2 + Tools

sudo apt update
sudo apt install -y \
  ros-humble-desktop \
  python3-colcon-common-extensions \
  python3-rosdep \
  git

# Initialize rosdep (only once per machine)
sudo rosdep init   # skip if already done
rosdep update

## Gazebo Sim (Ignition) + ROS bridge

sudo apt install -y \
  ros-humble-ros-gz-sim \
  ros-humble-ros-gz-bridge \
  ros-humble-ros-gz-image \
  ros-humble-ros-gz-interfaces

## Robot description & TF tools

sudo apt install -y \
  ros-humble-robot-state-publisher \
  ros-humble-joint-state-publisher \
  ros-humble-joint-state-publisher-gui \
  ros-humble-xacro \
  ros-humble-tf2-ros

## Control + Ackermann steering

sudo apt install -y \
  ros-humble-ros2-control \
  ros-humble-ros2-controllers \
  ros-humble-gz-ros2-control \
  ros-humble-ackermann-msgs \
  ros-humble-ackermann-steering-controller

## Localization, SLAM, Navigation, Teleop, RViz

sudo apt install -y \
  ros-humble-robot-localization \
  ros-humble-slam-toolbox \
  ros-humble-navigation2 \
  ros-humble-nav2-bringup \
  ros-humble-teleop-twist-keyboard \
  ros-humble-rviz2

## Optional: mirror full Python environment

# Only if you really want everything from requirements.txt
pip3 install -r requirements.txt
